37 bool IntervalsAreSortedAndNonAdjacent(
38  absl::Span<const ClosedInterval> intervals) {
39  for (int i = 1; i < intervals.size(); ++i) {
40  if (intervals[i - 1].start > intervals[i - 1].end) return false;
41  // First test make sure that intervals[i - 1].end + 1 will not overflow.
42  if (intervals[i - 1].end >= intervals[i].start ||
43  intervals[i - 1].end + 1 >= intervals[i].start) {
44  return false;
45  }
46  }
47  return intervals.empty() ? true
48  : intervals.back().start <= intervals.back().end;
49 }

62 // Transforms a sorted list of intervals in a sorted DISJOINT list for which
63 // IntervalsAreSortedAndNonAdjacent() would return true.
64 void UnionOfSortedIntervals(absl::InlinedVector<ClosedInterval, 1>* intervals) {
65  DCHECK(std::is_sorted(intervals->begin(), intervals->end()));
66  int new_size = 0;
67  for (const ClosedInterval& i : *intervals) {
68  if (new_size > 0 && i.start <= CapAdd((*intervals)[new_size - 1].end, 1)) {
69  (*intervals)[new_size - 1].end =
70  std::max(i.end, (*intervals)[new_size - 1].end);
71  } else {
72  (*intervals)[new_size++] = i;
73  }
74  }
75  intervals->resize(new_size);
76 
77  // This is important for InlinedVector in the case the result is a single
78  // intervals.
79  intervals->shrink_to_fit();
80  DCHECK(IntervalsAreSortedAndNonAdjacent(*intervals));
81 }

87  DCHECK_GT(positive_coeff, 0);

94  DCHECK_GT(positive_coeff, 0);
95  const int64 result = value / positive_coeff;
96  const int64 adjust = static_cast<int64>(result * positive_coeff > value);

221 bool Domain::Contains(int64 value) const {
222  // Because we only compare by start and there is no duplicate starts, this
223  // should be the next interval after the one that has a chance to contains
224  // value.
225  auto it = std::upper_bound(intervals_.begin(), intervals_.end(),
226  ClosedInterval(value, value));
227  if (it == intervals_.begin()) return false;
228  --it;
229  return value <= it->end;

233  int i = 0;

245 Domain Domain::Complement() const {
246  Domain result;
247  int64 next_start = kint64min;
248  result.intervals_.reserve(intervals_.size() + 1);
249  for (const ClosedInterval& interval : intervals_) {
250  if (interval.start != kint64min) {
251  result.intervals_.push_back({next_start, interval.start - 1});
252  }
253  if (interval.end == kint64max) return result;
254  next_start = interval.end + 1;
255  }
256  result.intervals_.push_back({next_start, kint64max});
257  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
258  return result;
259 }

267 void Domain::NegateInPlace() {
268  if (intervals_.empty()) return;
269  std::reverse(intervals_.begin(), intervals_.end());
270  if (intervals_.back().end == kint64min) {
271  // corner-case
272  intervals_.pop_back();
273  }
274  for (ClosedInterval& ref : intervals_) {
275  std::swap(ref.start, ref.end);
276  ref.start = ref.start == kint64min ? kint64max : -ref.start;
277  ref.end = ref.end == kint64min ? kint64max : -ref.end;
278  }
279  DCHECK(IntervalsAreSortedAndNonAdjacent(intervals_));
280 }

282 Domain Domain::IntersectionWith(const Domain& domain) const {
283  Domain result;
284  const auto& a = intervals_;
285  const auto& b = domain.intervals_;
286  for (int i = 0, j = 0; i < a.size() && j < b.size();) {
287  if (a[i].start <= b[j].start) {
288  if (a[i].end < b[j].start) {
289  // Empty intersection. We advance past the first interval.
290  ++i;
291  } else { // a[i].end >= b[j].start
292  // Non-empty intersection: push back the intersection of these two, and
293  // advance past the first interval to finish.
294  if (a[i].end <= b[j].end) {
295  result.intervals_.push_back({b[j].start, a[i].end});
296  ++i;
297  } else { // a[i].end > b[j].end.
298  result.intervals_.push_back({b[j].start, b[j].end});
299  ++j;
300  }
301  }
302  } else { // a[i].start > b[i].start.
303  // We do the exact same thing as above, but swapping a and b.
304  if (b[j].end < a[i].start) {
305  ++j;
306  } else { // b[j].end >= a[i].start
307  if (b[j].end <= a[i].end) {
308  result.intervals_.push_back({a[i].start, b[j].end});
309  ++j;
310  } else { // a[i].end > b[j].end.
311  result.intervals_.push_back({a[i].start, a[i].end});
312  ++i;
313  }
314  }
315  }
316  }
317  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
318  return result;
319 }

361 Domain Domain::MultiplicationBy(int64 coeff, bool* exact) const {
362  if (exact != nullptr) *exact = true;
363  if (intervals_.empty() || coeff == 0) return {};
364 
365  const int64 abs_coeff = std::abs(coeff);
366  const int64 size_if_non_trivial = abs_coeff > 1 ? Size() : 0;
367  if (size_if_non_trivial > kDomainComplexityLimit) {
368  if (exact != nullptr) *exact = false;
369  return ContinuousMultiplicationBy(coeff);
370  }
371 
372  Domain result;
373  if (abs_coeff > 1) {
374  const int64 max_value = kint64max / abs_coeff;
375  const int64 min_value = kint64min / abs_coeff;
376  result.intervals_.reserve(size_if_non_trivial);
377  for (const ClosedInterval& i : intervals_) {
378  for (int64 v = i.start;; ++v) {
379  // We ignore anything that overflow.
380  if (v >= min_value && v <= max_value) {
381  // Because abs_coeff > 1, all new values are disjoint.
382  const int64 new_value = v * abs_coeff;
383  result.intervals_.push_back({new_value, new_value});
384  }
385 
386  // This is to avoid doing ++v when v is kint64max!
387  if (v == i.end) break;
388  }
389  }
390  } else {
391  result = *this;
392  }
393  if (coeff < 0) result.NegateInPlace();
394  return result;
395 }

397 Domain Domain::ContinuousMultiplicationBy(int64 coeff) const {
398  Domain result = *this;
399  const int64 abs_coeff = std::abs(coeff);
400  for (ClosedInterval& i : result.intervals_) {
401  i.start = CapProd(i.start, abs_coeff);
402  i.end = CapProd(i.end, abs_coeff);
403  }
404  UnionOfSortedIntervals(&result.intervals_);
405  if (coeff < 0) result.NegateInPlace();
406  return result;
407 }

409 Domain Domain::ContinuousMultiplicationBy(const Domain& domain) const {
410  Domain result;
411  for (const ClosedInterval& i : this->intervals_) {
412  for (const ClosedInterval& j : domain.intervals_) {
413  ClosedInterval new_interval;
414  const int64 a = CapProd(i.start, j.start);
415  const int64 b = CapProd(i.end, j.end);
416  const int64 c = CapProd(i.start, j.end);
417  const int64 d = CapProd(i.end, j.start);
418  new_interval.start = std::min({a, b, c, d});
419  new_interval.end = std::max({a, b, c, d});
420  result.intervals_.push_back(new_interval);
421  }
422  }
423  std::sort(result.intervals_.begin(), result.intervals_.end());
424  UnionOfSortedIntervals(&result.intervals_);
425  return result;
426 }

441 Domain Domain::InverseMultiplicationBy(const int64 coeff) const {
442  if (coeff == 0) {
443  return Contains(0) ? Domain::AllValues() : Domain();
444  }
445  Domain result = *this;
446  int new_size = 0;
447  const int64 abs_coeff = std::abs(coeff);
448  for (const ClosedInterval& i : result.intervals_) {
449  const int64 start = CeilRatio(i.start, abs_coeff);
450  const int64 end = FloorRatio(i.end, abs_coeff);
451  if (start > end) continue;
452  if (new_size > 0 && start == result.intervals_[new_size - 1].end + 1) {
453  result.intervals_[new_size - 1].end = end;
454  } else {
455  result.intervals_[new_size++] = {start, end};
456  }
457  }
458  result.intervals_.resize(new_size);
459  result.intervals_.shrink_to_fit();
460  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
461  if (coeff < 0) result.NegateInPlace();
462  return result;
463 }

465 // It is a bit difficult to see, but this code is doing the same thing as
466 // for all interval in this.UnionWith(implied_domain.Complement())):
467 // - Take the two extreme points (min and max) in interval \inter implied.
468 // - Append to result [min, max] if these points exists.
469 Domain Domain::SimplifyUsingImpliedDomain(const Domain& implied_domain) const {
470  Domain result;
471  if (implied_domain.IsEmpty()) return result;
472 
473  int i = 0;
474  int64 min_point;
475  int64 max_point;
476  bool started = false;
477  for (const ClosedInterval interval : intervals_) {
478  // We only "close" the new result interval if it cannot be extended by
479  // implied_domain.Complement(). The only extension possible look like:
480  // interval_: ...] [....
481  // implied : ...] [... i ...]
482  if (started && implied_domain.intervals_[i].start < interval.start) {
483  result.intervals_.push_back({min_point, max_point});
484  started = false;
485  }
486 
487  // Find the two extreme points in interval \inter implied_domain.
488  // Always stop the loop at the first interval with and end strictly greater
489  // that interval.end.
490  for (; i < implied_domain.intervals_.size(); ++i) {
491  const ClosedInterval current = implied_domain.intervals_[i];
492  if (current.end >= interval.start && current.start <= interval.end) {
493  // Current and interval have a non-empty intersection.
494  const int64 inter_max = std::min(interval.end, current.end);
495  if (!started) {
496  started = true;
497  min_point = std::max(interval.start, current.start);
498  max_point = inter_max;
499  } else {
500  // No need to update the min_point here, and the new inter_max must
501  // necessarily be > old one.
502  DCHECK_GE(inter_max, max_point);
503  max_point = inter_max;
504  }
505  }
506  if (current.end > interval.end) break;
507  }
508  if (i == implied_domain.intervals_.size()) break;
509  }
510  if (started) {
511  result.intervals_.push_back({min_point, max_point});
512  }
513  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
514  return result;
515 }

526 bool Domain::operator<(const Domain& other) const {
527  const auto& d1 = intervals_;
528  const auto& d2 = other.intervals_;
529  const int common_size = std::min(d1.size(), d2.size());
530  for (int i = 0; i < common_size; ++i) {
531  const ClosedInterval& i1 = d1[i];
532  const ClosedInterval& i2 = d2[i];
533  if (i1.start < i2.start) return true;
534  if (i1.start > i2.start) return false;
535  if (i1.end < i2.end) return true;
536  if (i1.end > i2.end) return false;
537  }
538  return d1.size() < d2.size();

543 int64 SumOfKMinValueInDomain(const Domain& domain, int k) {
544  int64 current_sum = 0.0;
545  int current_index = 0;
546  for (const ClosedInterval interval : domain) {
547  if (current_index >= k) break;
548  for (int v(interval.start); v <= interval.end; ++v) {

580 SortedDisjointIntervalList
581 SortedDisjointIntervalList::BuildComplementOnInterval(int64 start, int64 end) {
582  SortedDisjointIntervalList interval_list;
583  int64 next_start = start;
584  for (auto it = FirstIntervalGreaterOrEqual(start); it != this->end(); ++it) {
585  const ClosedInterval& interval = *it;
586  const int64 next_end = CapSub(interval.start, 1);
587  if (next_end > end) break;
588  if (next_start <= next_end) {
589  interval_list.InsertInterval(next_start, next_end);
590  }
591  next_start = CapAdd(interval.end, 1);
592  }
593  if (next_start <= end) {
594  interval_list.InsertInterval(next_start, end);
595  }
596  return interval_list;
597 }

599 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::InsertInterval(
600  int64 start, int64 end) {
601  // start > end could mean an empty interval, but we prefer to LOG(DFATAL)
602  // anyway. Really, the user should never give us that.
603  if (start > end) {
604  LOG(DFATAL) << "Invalid interval: " << ClosedInterval({start, end});
605  return intervals_.end();
606  }
607 
608  auto result = intervals_.insert({start, end});
609  if (!result.second) return result.first; // Duplicate: exit immediately.
610 
611  // TODO(user): tune the algorithm below if it proves to be a bottleneck.
612  // For example, one could try to avoid an insertion if it's not needed
613  // (when the interval merges with a single existing interval or is fully
614  // contained by one).
615 
616  // Iterate over the previous iterators whose end is after (or almost at) our
617  // start. After this, "it1" will point to the first interval that needs to be
618  // merged with the current interval (possibly pointing to the current interval
619  // itself, if no "earlier" interval should be merged).
620  auto it1 = result.first;
621  if (start == kint64min) { // Catch underflows
622  it1 = intervals_.begin();
623  } else {
624  const int64 before_start = start - 1;
625  while (it1 != intervals_.begin()) {
626  auto prev_it = it1;
627  --prev_it;
628  if (prev_it->end < before_start) break;
629  it1 = prev_it;
630  }
631  }
632 
633  // Ditto, on the other side: "it2" will point to the interval *after* the last
634  // one that should be merged with the current interval.
635  auto it2 = result.first;
636  if (end == kint64max) {
637  it2 = intervals_.end();
638  } else {
639  const int64 after_end = end + 1;
640  do {
641  ++it2;
642  } while (it2 != intervals_.end() && it2->start <= after_end);
643  }
644 
645  // [it1..it2) is the range (inclusive on it1, exclusive on it2) of intervals
646  // that should be merged together. We'll set it3 = it2-1 and erase [it1..it3)
647  // and set *it3 to the merged interval.
648  auto it3 = it2;
649  it3--;
650  if (it1 == it3) return it3; // Nothing was merged.
651  const int64 new_start = std::min(it1->start, start);
652  const int64 new_end = std::max(it3->end, end);
653  auto it = intervals_.erase(it1, it3);
654  // HACK(user): set iterators point to *const* values. Which is expected,
655  // because if one alters a set element's value, then it collapses the set
656  // property! But in this very special case, we know that we can just overwrite
657  // it->start, so we do it.
658  const_cast<ClosedInterval*>(&(*it))->start = new_start;
659  const_cast<ClosedInterval*>(&(*it))->end = new_end;
660  return it;
661 }

663 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::GrowRightByOne(
664  int64 value, int64* newly_covered) {
665  auto it = intervals_.upper_bound({value, kint64max});
666  auto it_prev = it;
667 
668  // No interval containing or adjacent to "value" on the left (i.e. below).
669  if (it != begin()) {
670  --it_prev;
671  }
672  if (it == begin() || ((value != kint64min) && it_prev->end < value - 1)) {
673  *newly_covered = value;
674  if (it == end() || it->start != value + 1) {
675  // No interval adjacent to "value" on the right: insert a singleton.
676  return intervals_.insert(it, {value, value});
677  } else {
678  // There is an interval adjacent to "value" on the right. Extend it by
679  // one. Note that we already know that there won't be a merge with another
680  // interval on the left, since there were no interval adjacent to "value"
681  // on the left.
682  DCHECK_EQ(it->start, value + 1);
683  const_cast<ClosedInterval*>(&(*it))->start = value;
684  return it;
685  }
686  }
687 
688  // At this point, "it_prev" points to an interval containing or adjacent to
689  // "value" on the left: grow it by one, and if it now touches the next
690  // interval, merge with it.
691  CHECK_NE(kint64max, it_prev->end) << "Cannot grow right by one: the interval "
692  "that would grow already ends at "
693  "kint64max";
694  *newly_covered = it_prev->end + 1;
695  if (it != end() && it_prev->end + 2 == it->start) {
696  // We need to merge it_prev with 'it'.
697  const_cast<ClosedInterval*>(&(*it_prev))->end = it->end;
698  intervals_.erase(it);
699  } else {
700  const_cast<ClosedInterval*>(&(*it_prev))->end = it_prev->end + 1;
701  }
702  return it_prev;
703 }
