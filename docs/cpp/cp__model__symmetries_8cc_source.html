69 // Returns a graph whose automorphisms can be mapped back to the symmetries of
70 // the model described in the given CpModelProto.
71 //
72 // Any permutation of the graph that respects the initial_equivalence_classes
73 // output can be mapped to a symmetry of the given problem simply by taking its
74 // restriction on the first num_variables nodes and interpreting its index as a
75 // variable index. In a sense, a node with a low enough index #i is in
76 // one-to-one correspondence with the variable #i (using the index
77 // representation of variables).
78 //
79 // The format of the initial_equivalence_classes is the same as the one
80 // described in GraphSymmetryFinder::FindSymmetries(). The classes must be dense
81 // in [0, num_classes) and any symmetry will only map nodes with the same class
82 // between each other.
83 template <typename Graph>
84 std::unique_ptr<Graph> GenerateGraphForSymmetryDetection(
85  const CpModelProto& problem,
86  std::vector<int>* initial_equivalence_classes) {
87  CHECK(initial_equivalence_classes != nullptr);
88 
89  const int num_variables = problem.variables_size();
90  auto graph = absl::make_unique<Graph>();
91  initial_equivalence_classes->clear();
92  enum NodeType {
93  VARIABLE_NODE,
94  CONSTRAINT_NODE,
95  CONSTRAINT_COEFFICIENT_NODE,
96  ENFORCEMENT_LITERAL
97  };
98  IdGenerator id_generator;
99 
100  // For two variables to be in the same equivalence class, they need to have
101  // the same objective coefficient, and the same possible bounds.
102  std::vector<int64> objective_by_var(num_variables, 0);
103  for (int i = 0; i < problem.objective().vars_size(); ++i) {
104  objective_by_var[problem.objective().vars(i)] =
105  problem.objective().coeffs(i);
106  }
107 
108  auto new_node = [&initial_equivalence_classes,
109  &id_generator](const std::vector<int64>& key) {
110  // Since we add nodes one by one, initial_equivalence_classes->size() gives
111  // the number of nodes at any point, which we use as the next node index.
112  const int node = initial_equivalence_classes->size();
113  initial_equivalence_classes->push_back(id_generator.GetId(key));
114  return node;
115  };
116 
117  for (int v = 0; v < num_variables; ++v) {
118  const IntegerVariableProto& variable = problem.variables(v);
119  std::vector<int64> key = {VARIABLE_NODE, objective_by_var[v]};
120  Append(variable.domain(), &key);
121  CHECK_EQ(v, new_node(key));
122  // Make sure the graph contains all the variable nodes, even if no edges are
123  // attached to them through constraints.
124  graph->AddNode(v);
125  }
126 
127  auto add_edge = [&graph](int node_1, int node_2) {
128  graph->AddArc(node_1, node_2);
129  graph->AddArc(node_2, node_1);
130  };
131  auto add_literal_edge = [&add_edge, &new_node](int ref, int constraint_node) {
132  const int variable_node = PositiveRef(ref);
133  if (RefIsPositive(ref)) {
134  add_edge(variable_node, constraint_node);
135  } else {
136  const int coefficient_node = new_node({CONSTRAINT_COEFFICIENT_NODE, -1});
137 
138  add_edge(variable_node, coefficient_node);
139  add_edge(coefficient_node, constraint_node);
140  }
141  };
142 
143  // Add constraints to the graph.
144  for (const ConstraintProto& constraint : problem.constraints()) {
145  const int constraint_node = initial_equivalence_classes->size();
146  std::vector<int64> key = {CONSTRAINT_NODE, constraint.constraint_case()};
147 
148  switch (constraint.constraint_case()) {
149  case ConstraintProto::kLinear: {
150  Append(constraint.linear().domain(), &key);
151  CHECK_EQ(constraint_node, new_node(key));
152 
153  for (int i = 0; i < constraint.linear().vars_size(); ++i) {
154  if (constraint.linear().coeffs(i) == 0) continue;
155  const int ref = constraint.linear().vars(i);
156  const int variable_node = PositiveRef(ref);
157  const int64 coeff = RefIsPositive(ref)
158  ? constraint.linear().coeffs(i)
159  : -constraint.linear().coeffs(i);
160  if (coeff == 1) {
161  // For all coefficients equal to one, which are the most common, we
162  // can optimize the size of the graph by omitting the coefficient
163  // node altogether.
164  add_edge(variable_node, constraint_node);
165  } else {
166  const int coefficient_node =
167  new_node({CONSTRAINT_COEFFICIENT_NODE, coeff});
168 
169  add_edge(variable_node, coefficient_node);
170  add_edge(coefficient_node, constraint_node);
171  }
172  }
173  break;
174  }
175  case ConstraintProto::kBoolOr: {
176  CHECK_EQ(constraint_node, new_node(key));
177  for (const int ref : constraint.bool_or().literals()) {
178  add_literal_edge(ref, constraint_node);
179  }
180  break;
181  }
182  case ConstraintProto::kBoolXor: {
183  CHECK_EQ(constraint_node, new_node(key));
184  for (const int ref : constraint.bool_xor().literals()) {
185  add_literal_edge(ref, constraint_node);
186  }
187  break;
188  }
189  case ConstraintProto::kBoolAnd: {
190  if (constraint.enforcement_literal_size() == 0) {
191  // All literals are true in this case.
192  CHECK_EQ(constraint_node, new_node(key));
193  for (const int ref : constraint.bool_and().literals()) {
194  add_literal_edge(ref, constraint_node);
195  }
196  break;
197  }
198 
199  // To make the BoolAnd constraint more generic in the graph, we expand
200  // it into a set of BoolOr constraints where
201  // not(enforcement_literal) OR literal = true
202  // for all constraint's literals. This is equivalent to
203  // enforcement_literal => literal
204  // for all literals.
205  std::vector<int64> key = {CONSTRAINT_NODE, ConstraintProto::kBoolOr};
206  const int non_enforcement_literal =
207  NegatedRef(constraint.enforcement_literal(0));
208  for (const int literal : constraint.bool_and().literals()) {
209  const int constraint_node = new_node(key);
210  add_literal_edge(non_enforcement_literal, constraint_node);
211  add_literal_edge(literal, constraint_node);
212  }
213  break;
214  }
215  default: {
216  // If the model contains any non-supported constraints, return an empty
217  // graph.
218  // TODO(user): support other types of constraints.
219  LOG(ERROR) << "Unsupported constraint type "
220  << constraint.constraint_case();
221  return nullptr;
222  }
223  }
224 
225  if (constraint.constraint_case() != ConstraintProto::kBoolAnd &&
226  constraint.enforcement_literal_size() > 0) {
227  const int ref = constraint.enforcement_literal(0);
228  const int enforcement_literal_node = PositiveRef(ref);
229  const int enforcement_type_node =
230  new_node({ENFORCEMENT_LITERAL, RefIsPositive(ref)});
231 
232  add_edge(constraint_node, enforcement_type_node);
233  add_edge(enforcement_type_node, enforcement_literal_node);
234  }
235  }
236 
237  graph->Build();
238  DCHECK_EQ(graph->num_nodes(), initial_equivalence_classes->size());
239  return graph;
240 }

243 void FindCpModelSymmetries(
244  const CpModelProto& problem,
245  std::vector<std::unique_ptr<SparsePermutation>>* generators,
246  double time_limit_seconds) {
247  CHECK(generators != nullptr);
248  generators->clear();
249 
250  typedef GraphSymmetryFinder::Graph Graph;
251 
252  std::vector<int> equivalence_classes;
253  std::unique_ptr<Graph> graph(
254  GenerateGraphForSymmetryDetection<Graph>(problem, &equivalence_classes));
255  if (graph == nullptr) return;
256 
257  LOG(INFO) << "Graph has " << graph->num_nodes() << " nodes and "
258  << graph->num_arcs() / 2 << " edges.";
259 
260  GraphSymmetryFinder symmetry_finder(*graph, /*is_undirected=*/true);
261  std::vector<int> factorized_automorphism_group_size;
262  CHECK_OK(symmetry_finder.FindSymmetries(time_limit_seconds,
263  &equivalence_classes, generators,
264  &factorized_automorphism_group_size));
265 
266  // Remove from the permutations the part not concerning the variables.
267  // Note that some permutations may become empty, which means that we had
268  // duplicate constraints.
269  double average_support_size = 0.0;
270  int num_generators = 0;
271  for (int i = 0; i < generators->size(); ++i) {
272  SparsePermutation* permutation = (*generators)[i].get();
273  std::vector<int> to_delete;
274  for (int j = 0; j < permutation->NumCycles(); ++j) {
275  // Because variable nodes are in a separate equivalence class than any
276  // other node, a cycle can either contain only variable nodes or none, so
277  // we just need to check one element of the cycle.
278  if (*(permutation->Cycle(j).begin()) >= problem.variables_size()) {
279  to_delete.push_back(j);
280  if (DEBUG_MODE) {
281  // Verify that the cycle's entire support does not touch any variable.
282  for (const int node : permutation->Cycle(j)) {
283  DCHECK_GE(node, problem.variables_size());
284  }
285  }
286  }
287  }
288  permutation->RemoveCycles(to_delete);
289  if (!permutation->Support().empty()) {
290  average_support_size += permutation->Support().size();
291  swap((*generators)[num_generators], (*generators)[i]);
292  ++num_generators;
293  } else {
294  LOG(INFO) << "The model contains duplicate constraints!";
295  }
296  }
297  generators->resize(num_generators);
298  average_support_size /= num_generators;
299  LOG(INFO) << "# of generators: " << num_generators;
300  LOG(INFO) << "Average support size: " << average_support_size;
301 }
